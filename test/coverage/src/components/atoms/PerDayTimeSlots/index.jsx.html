const DayList = ({
  setSelectedDayToApplyTime,
  day,
  selectedDayToApplyTime,
  applyTimeToSelectedDay,
  dayListEnabled,
}) => {
  const { t } = useTranslation();
  const screen = useBreakpoint();
  const md = screen?.md ?? true;
 
  return (
    <>
      <OptionList
        onChange={setSelectedDayToApplyTime}
        title={t('COPY_TIMES_TO')}
        sections={[
          {
            options: Object.values(days).map((eachDay) => ({
              value: eachDay,
              label: t(eachDay),
              disabled: md ? eachDay === day || !dayListEnabled[eachDay] : eachDay === day,

          },

        selected={selectedDayToApplyTime}
        allowMultiple

          </div>
        </Space>
      ) : null}
    </>
  );
