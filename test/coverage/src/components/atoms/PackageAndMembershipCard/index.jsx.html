import React from 'react';
 
import { Space } from 'antd';
import { BookOutlined, CheckCircleFilled, StarOutlined } from '@ant-design/icons';
import moment from 'moment';
import { useTranslation } from 'react-i18next';
 
import styles from './styles.module.scss';
 
export const PackageAndMembershipCard = ({ perPackage, membership }) => {
  const { t } = useTranslation();
  return (
    <div className={styles.packageAndMembershipCard}>
      <div className={styles.packageAndMembershipContentWrapper}>
        <div className={styles.packageAndMembershipServicesNameContainer}>
          {perPackage ? <BookOutlined className={styles.starIcon} /> : <StarOutlined className={styles.starIcon} />}
          <p className={styles.packageAndMembershipServicesName}>{perPackage ? perPackage?.name : membership?.name}</p>
        </div>
        <Space className={styles.membershipInfoContainer}>
          <div className={styles.membershipStatusContainer}>
            <CheckCircleFilled className={styles.CheckCircle} />
            <p style={{ fontSize: 13 }}>
              {`${t(perPackage ? perPackage?.status : membership?.status)} | ${t('NEXT_BILLING')}: ${moment(
                perPackage ? perPackage?.expiry_date : membership?.expiry_date
              ).format('DD MMM YYYY')}`}
            </p>
          </div>
        </Space>
      </div>
      <div className={styles.contentLabelContainer}>
        <p className={membership ? styles.contentLabelMembership : styles.contentLabelPackage}>
          {perPackage ? 'PACKAGE' : null}
          {membership ? 'MEMBERSHIP' : null}
        </p>
      </div>
    </div>
  );
};
