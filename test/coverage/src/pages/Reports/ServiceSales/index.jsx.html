import React, { useCallback, useEffect, useState } from 'react';
 
import moment from 'moment';
import { Divider, Empty, Grid, Select, Space, Spin } from 'antd';
import { Page, Layout, LegacyCard } from '@shopify/polaris';
import { DownloadOutlined } from '@ant-design/icons';
import { useTranslation } from 'react-i18next';
 
import { CommonButton, ReportTable, DatePickerLocale } from '@/components';
 
import { useAuthorizedQuery, useReport } from '@/hooks';
import { API_PATH, API_REQUEST_METHODS, REPORTS_SHOW_BY } from '@/constants';
import { dateHelper, formatDateMonthYear, formatEndOfTodayDate, formatStartOfTodayDate, formatUTC } from '@/utils';
import { useAppContext } from '@/context/AppContext';
 
import styles from './styles.module.scss';
 
const { useBreakpoint } = Grid;
const { formatDate } = dateHelper;
 
export const ServiceSales = () => {
  const screen = useBreakpoint();
  const md = screen?.md ?? true;
  const { storeExternalId } = useAppContext();
  const { downloadTable, generateRenderColumns } = useReport();
  const { t } = useTranslation();
 
  const [report, setReport] = useState(null);
  const [filterForm, setFilterForm] = useState({
    reportRange: [],
    show_by: [],
  });
 
  const rangePresets = [
    {
      label: t('LAST_X_DAYS', [7]),
      value: [moment().add(-7, 'd'), moment()],
    },
    {
      label: t('LAST_X_DAYS', [14]),
      value: [moment().add(-14, 'd'), moment()],
    },
    {
      label: t('LAST_X_DAYS', [30]),
      value: [moment().add(-30, 'd'), moment()],
    },
  ];
 
  const disabledDate = useCallback((current) => {
    // add this so user will not able to select days after today
    return current && current > moment().endOf('day');
  }, []);
 
  const handleSelectReportsRange = useCallback((value, field) => {
    setFilterForm((prevValue) => {
      let temp = { ...prevValue };
      temp[field] = value;
 
      return temp;
    });
  }, []);
 
  const generatePayload = () => {
    const start_date = formatUTC(formatStartOfTodayDate(filterForm?.reportRange[0]));
    const end_date = formatUTC(formatEndOfTodayDate(filterForm?.reportRange[1]));
    const showBy = filterForm?.show_by;
    return {
      start_date,
      end_date,
      show_by: showBy,
    };
  };
 
  const {
    data: reportData,
    refetch: generate,
    isLoading,
  } = useAuthorizedQuery({
    urlPath: API_PATH.SECURE.REPORTS.STAFF.SERVICE_SALES(storeExternalId, generatePayload()),
    method: API_REQUEST_METHODS.GET,
    reactQueryOptions: {
      enabled: false,
      cacheTime: 0,
    },
  });
 
  useEffect(() => {
    if (reportData) {
      const data = reportData?.data?.map((el) => ({
        service: el.service,
        ...el,
      }));
 
      setReport({ data, footer: { ...reportData?.footer, service: { value: t('TOTAL') } } });
    }
  }, [reportData]);
 
  return (
    <div>
      <Page fullWidth title={t('SERVICE_SALES')} subtitle={md ? t('REPORT_SUBTITLE') : null}>
        <Layout>
          <Layout.Section>
            <LegacyCard sectioned>
              <Space direction="vertical" size={28} style={{ width: 290 }}>
                <div>
                  <p>{t('SELECT_YOUR_CUSTOM_PERIOD')}</p>
                  <DatePickerLocale
                    type="rangePicker"
                    // disabledDate={disabledDate}
                    style={{ width: 290 }}
                    value={filterForm?.reportRange}
                    onChange={(val) => handleSelectReportsRange(val, 'reportRange')}
                    disabled={isLoading}
                    allowClear={false}
                    presets={rangePresets}
                  />
                </div>
                <div>
                  <p>{t('SHOW_BY')}</p>
                  <Select
                    placeholder={t('SHOW_BY')}
                    value={filterForm?.show_by}
                    onChange={(val) => handleSelectReportsRange(val, 'show_by')}
                    style={{ width: '100%' }}
                    options={REPORTS_SHOW_BY}
                    disabled={isLoading}
                  />
                </div>
                <CommonButton
                  onClick={generate}
                  disabled={filterForm?.reportRange?.length === 0 || filterForm?.show_by?.length === 0 || isLoading}
                  fullWidth
                  primary
                >
                  {t('GENERATE')}
                </CommonButton>
              </Space>
 
              {report ? (
                <>
                  <Divider />
 
                  {report?.data?.length === 0 ? (
                    <Empty />
                  ) : (
                    <Spin tip={t('LOADING')} spinning={isLoading}>
                      <div className={styles.tableInfoContainer}>
                        {filterForm?.reportRange?.length === 2 ? (
                          <p style={{ marginBottom: 20 }}>
                            {t('PERIOD_X_X', [
                              formatDateMonthYear(filterForm?.reportRange[0]),
                              formatDateMonthYear(filterForm?.reportRange[1]),
                            ])}
                          </p>
                        ) : null}
                        <CommonButton
                          icon={<DownloadOutlined />}
                          onClick={() => downloadTable(report ?? [], 'report ' + formatDate.toShortDate(moment()))}
                        >
                          {t('EXPORT_TO_EXCEL')}
                        </CommonButton>
                      </div>
                      <ReportTable
                        columns={generateRenderColumns(report, 'service')}
                        data={report ?? []}
                        title={''}
                        firstColumn=""
                      />
                    </Spin>
                  )}
                </>
              ) : null}
            </LegacyCard>
          </Layout.Section>
        </Layout>
      </Page>
    </div>
  );
};
